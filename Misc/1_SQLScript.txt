--Tabelle mit Detailinhalten zu den LVs
--Verbindung aufbauen
--Resultset abrufen
--DML-Operationen ausführen
CREATE TABLE Detailinhalte
(
    InhaltNR INT PRIMARY KEY,
    LVNr INT, --Foreign Key Constraint (Primary Key, Foreign Key, Check, Not null)
    Inhalt VARCHAR(50)
);

--Constraint bestimmt gültigkeit von Daten
--Primary Key
